CREATE TABLE authors (
    authors_id SERIAL PRIMARY KEY,
    authors_name TEXT NOT NULL,
);

CREATE TABLE books (
    book_id SERIAL PRIMARY KEY,
    title TEXT NOT NULL,
    published_year INT,
    author_id INT NOT NULL,
    Foreign Key (author_id) REFERENCES authors (author_id)
);
